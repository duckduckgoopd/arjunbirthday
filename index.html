  /* ── Recording ── */

  async function toggleRecord() {

    const name = document.getElementById('name-input').value.trim();

    if (!name) {

      document.getElementById('voice-name-reminder').style.display = 'block';

      document.getElementById('name-input').focus();

      return;

    }

    document.getElementById('voice-name-reminder').style.display = 'none';

    if (!isRecording) {

      try {

        const stream = await navigator.mediaDevices.getUserMedia({ audio: true });

        mediaRecorder = new MediaRecorder(stream);

        audioChunks = [];

        mediaRecorder.ondataavailable = e => audioChunks.push(e.data);

        mediaRecorder.onstop = () => {

          recordedBlob = new Blob(audioChunks, { type: 'audio/webm' });

          const url = URL.createObjectURL(recordedBlob);

          const a = document.getElementById('recorded-audio');

          a.src = url; a.style.display = 'block';

          uploadedFile = null;

          document.getElementById('uploaded-audio').style.display = 'none';

          uploadedAudioURL = null;

        };

        mediaRecorder.start();

        isRecording = true; seconds = 0;

        document.getElementById('rec-btn').classList.add('recording');

        document.getElementById('rec-status').textContent = 'Recording… tap to stop';

        document.getElementById('rec-timer').style.display = 'block';

        document.getElementById('waveform').classList.add('active');

        timerInterval = setInterval(() => {

          seconds++;

          const m = Math.floor(seconds/60), s = seconds%60;

          document.getElementById('rec-timer').textContent = m+':'+(s<10?'0':'')+s;

        }, 1000);

      } catch(e) {

        document.getElementById('rec-status').textContent = 'Mic unavailable — please use the upload option below.';

      }

    } else {

      mediaRecorder.stop();

      mediaRecorder.stream.getTracks().forEach(t => t.stop());

      isRecording = false;

      clearInterval(timerInterval);

      document.getElementById('rec-btn').classList.remove('recording');

      document.getElementById('rec-status').textContent = 'Tap to record again';

      document.getElementById('waveform').classList.remove('active');

    }

  }

 

  function handleUpload(e) {

    const name = document.getElementById('name-input').value.trim();

    if (!name) {

      document.getElementById('voice-name-reminder').style.display = 'block';

      document.getElementById('name-input').focus();

      e.target.value = '';

      return;

    }

    document.getElementById('voice-name-reminder').style.display = 'none';

    const file = e.target.files[0];

    if (!file) return;

    uploadedFile = file;

    recordedBlob = null;

    uploadedAudioURL = null;

    document.getElementById('recorded-audio').style.display = 'none';

    const a = document.getElementById('uploaded-audio');

    a.src = URL.createObjectURL(file);

    a.style.display = 'block';

    document.getElementById('upload-label').textContent = '✓ ' + file.name;

  }

 

  /* Upload audio to Uploadcare - works from any browser, no CORS issues */

  async function uploadAudio(blob, filename) {

    const prog = document.getElementById('upload-progress');

    prog.style.display = 'block';

    prog.textContent = 'Uploading audio...';

 

    const fd = new FormData();

    fd.append('UPLOADCARE_PUB_KEY', 'e4798ac19378af6c9e54');

    fd.append('UPLOADCARE_STORE', '1');

    fd.append('file', blob, filename);

 

    const res = await fetch('https://upload.uploadcare.com/base/', {

      method: 'POST', body: fd

    });

    if (!res.ok) throw new Error('Audio upload failed');

    const data = await res.json();

    const fileUrl = 'https://ucarecdn.com/' + data.file + '/';

    prog.textContent = 'Audio uploaded!';

    return fileUrl;

  }

 

  /* ── Submit ── */

  async function submitForm() {

    const name = document.getElementById('name-input').value.trim();

    const note = document.getElementById('note-input').value.trim();

    const songs = Array.from(document.querySelectorAll('.song-row input')).map(i=>i.value.trim()).filter(Boolean);

    const errEl = document.getElementById('error-msg');

    errEl.style.display = 'none';

 

    if (!name) { showError('Please enter your name!'); return; }

    if (songs.length === 0) { showError('Please suggest at least one song!'); return; }

 

    const btn = document.getElementById('submit-btn');

    btn.disabled = true; btn.textContent = 'Sending…';

 

    try {

      // Upload audio first if present

      const audioBlob = recordedBlob || uploadedFile;

      let audioURL = '';

      if (audioBlob) {

        const ext = uploadedFile ? uploadedFile.name.split('.').pop() : 'webm';

        const fname = `message-from-${name.replace(/\s+/g,'-')}.${ext}`;

        audioURL = await uploadAudio(audioBlob, fname);

      }

 

      // Send text data to Formspree

      const formData = new FormData();

      formData.append('_subject', `🎶 Mixtape contribution from ${name} for Arjun S.`);

      formData.append('From', name);

      formData.append('Songs', songs.join('\n'));

      if (note) formData.append('Note', note);

      if (audioURL) formData.append('Voice message (link)', audioURL);

 

      const res = await fetch('https://formspree.io/f/xnjoazqa', {

        method: 'POST', body: formData,

        headers: { 'Accept': 'application/json' }

      });

 

      if (res.ok) {

        document.getElementById('form-section').style.display = 'none';

        document.getElementById('success-screen').style.display = 'block';

        window.scrollTo({ top: 0, behavior: 'smooth' });

      } else {

        const d = await res.json();

        throw new Error(d.error || 'Formspree error');

      }

    } catch(e) {

      showError('Something went wrong — please check your connection and try again. (' + e.message + ')');

      btn.disabled = false;

      btn.textContent = 'Put me on the tape ♫';

      document.getElementById('upload-progress').style.display = 'none';

    }

  }

 

  function showError(msg) {

    const el = document.getElementById('error-msg');

  }
